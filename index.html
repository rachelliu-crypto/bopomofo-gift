    <!-- 輪播控制 & 複製代碼 JavaScript -->

    <script>

        let currentSlide = 0;

        const totalSlides = 4;

        let autoPlayInterval;

        let copyTimeouts = {}; // 用於獨立管理各幻燈片的複製還原定時器


        // 複製代碼功能 (支援 iframe 安全限制，並配備原地局部回饋)

        function copyCouponCode(text, element) {

            const el = document.createElement("textarea");

            el.value = text;

            el.style.position = 'fixed';

            el.style.opacity = '0';

            document.body.appendChild(el);

            el.select();

            

            try {

                const successful = document.execCommand('copy');

                if (successful) {

                    const icon = element.querySelector(".copy-icon");

                    const statusText = element.querySelector(".copy-text");


                    if (icon && statusText) {

                        // 轉換為精美的成功綠色打勾樣式

                        icon.className = "copy-icon fa-solid fa-check text-emerald-500 text-lg scale-110 transition-all duration-200";

                        statusText.textContent = "複製成功！";

                        statusText.className = "copy-text text-[9px] text-emerald-600 font-bold mt-1 scale-90 whitespace-nowrap transition-all duration-200";


                        const slideIndex = element.getAttribute('data-slide-index') || 'default';

                        

                        // 清除該卡片上一次未完成的還原定時，避免衝突

                        if (copyTimeouts[slideIndex]) {

                            clearTimeout(copyTimeouts[slideIndex]);

                        }


                        // 2 秒後還原為初始樣式

                        copyTimeouts[slideIndex] = setTimeout(() => {

                            icon.className = "copy-icon fa-solid fa-copy text-[#483a7a] text-lg group-hover:scale-110 transition-all duration-200";

                            statusText.textContent = "點擊複製";

                            statusText.className = "copy-text text-[9px] text-slate-500 font-bold mt-1 scale-90 whitespace-nowrap transition-all duration-200";

                        }, 2000);

                    }

                }

            } catch (err) {

                console.error('複製失敗:', err);

            } finally {

                document.body.removeChild(el);

            }

        }


        function updateCarousel() {

            const track = document.getElementById('carouselTrack');

            if (track) {

                track.style.transform = `translateX(-${currentSlide * 100}%)`;

            }

            

            const dots = document.querySelectorAll('#carouselDots span');

            dots.forEach((dot, index) => {

                if (index === currentSlide) {

                    dot.classList.remove('bg-gray-300');

                    dot.classList.add('bg-[#483a7a]', 'w-3');

                } else {

                    dot.classList.remove('bg-[#483a7a]', 'w-3');

                    dot.classList.add('bg-gray-300', 'w-1.5');

                }

            });

        }


        function nextSlide() {

            currentSlide = (currentSlide + 1) % totalSlides;

            updateCarousel();

        }


        function prevSlide() {

            currentSlide = (currentSlide - 1 + totalSlides) % totalSlides;

            updateCarousel();

        }


        function goToSlide(slideIndex) {

            currentSlide = slideIndex;

            updateCarousel();

        }


        function startAutoPlay() {

            autoPlayInterval = setInterval(nextSlide, 3000);

        }


        function stopAutoPlay() {

            clearInterval(autoPlayInterval);

        }


        window.onload = function () {

            updateCarousel();

            startAutoPlay();


            // 當滑鼠移入「精選線上課程」大卡片時暫停自動播放

            const courseCardElement = document.querySelector('.bento-card.shadow-classic.pt-10');

            if (courseCardElement) {

                courseCardElement.addEventListener('mouseenter', stopAutoPlay);

                courseCardElement.addEventListener('mouseleave', startAutoPlay);

            }

        }

    </script>
